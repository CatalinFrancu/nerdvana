dec_x:
  jeq X zero end_prog
  dec X

# add Y to Out, make i count from 0 to Y
pass_1:
  jeq i Y end_pass_1
  inc i
  inc Out
  jeq X X pass_1
end_pass_1:

dec_x_2:
  jeq X zero end_prog
  dec X

# add Y to Out, make i count from Y to 0
pass_2:
  jeq i zero end_pass_2
  dec i
  inc Out
  jeq X X pass_2
end_pass_2:

  jeq X X dec_x

end_prog:
