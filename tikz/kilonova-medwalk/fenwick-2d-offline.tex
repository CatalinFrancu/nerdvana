\usetikzlibrary{matrix}

\newcommand\FenwickTree[2]{ % anchor, data rows
  \matrix[
  at=(#1),
  anchor=north,
  draw,
  matrix of nodes,
  ampersand replacement=\&,
  column sep=3pt,
  nodes={minimum height=2em, text width=1cm, align=center, anchor=base},
  ]{
    \node{pos}; \& \node{AIB}; \\
    \node{...}; \& \node{...}; \\
    #2
    \node{...}; \& \node{...}; \\
  };
}
